**Commands**

# Setup ibus-bamboo
  $ sudo add-apt-repository ppa:bamboo-engine/ibus-bamboo
  $ sudo apt-get update
  $ sudo apt-get install ibus-bamboo
  $ ibus restart
# Setup GIT
	$ sudo add-apt-repository ppa:git-core/ppa
	$ sudo apt update
	$ sudo apt install git
# Setup OpenJDK 8
	$ sudo apt install openjdk-8-jdk
